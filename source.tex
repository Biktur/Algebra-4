\begin{theorem}[Машке]
$k$ --- поле, $G$ --- конечная группа, $\chr k \nmid |G|$, $V$ --- $k[G]$-модуль, $U$ -- подмодуль. Тогда найдётся подмодуль $W$, т. ч.  $V = U \oplus W$.
\end{theorem}

\begin{definition}
\textit{Неприводимый $G$-модуль} --- простой $k[G]$-модуль.
\end{definition}

\begin{definition}
Модуль $M$ называется \textit{неразложимым}, если равенство $M = M_1 \oplus M_2$, означает, что $M_1$ или $M_2$ тривиален.
\end{definition}

Везде далее $V$ --- $k[G]$-модуль, который рассматривается как конечномерное векторное пространство над $k$ (стоит отметить, что $|G| = \dim V < \infty$), а также $\chr k \nmid |G|$. При таких условиях теорему Машке можно переформулировать следующим образом:

\begin{theorem}[Машке]
    $V$ --- приводимый модуль $\Ra$ $V = V' \oplus V''$, где $V', V'' \neq 0$.
\end{theorem}

Понятно, что размерности $V', V''$ как векторных пространств над $k$ строго меньше чем размерность $V$. Также понятно, что если $V'$ или $V''$ являются приводимыми, то к ним снова можно применить теорему. Таким образом, для всякого $G$-модуля $V$ получается разложение $V = V_1 \oplus V_2 \oplus \hdots \oplus V_m$, где $\{V_i\}$ --- неприводимые модули.

\begin{lemma}[Шур]
Пусть $M, N$ --- простые $R$-модули. Тогда
\begin{enumerate}
\item Любой ненулевой гомоморфизм $M \xrightarrow{\varphi} N$ является изоморфизмом.
\item $\End_k(M)$ --- тело (т. е. кольцо с делением).
\end{enumerate}
\end{lemma}
\begin{proof}
\begin{enumerate}
\item Пусть $\varphi$ не изоморфизм. Тогда $\ker \varphi \neq 0$, или $\im \varphi \neq N$. В первом случае $\ker \varphi = M$, во втором --- $\im \varphi = 0$, т. е. $\varphi = 0$.
\item Ну действительно, любой ненулевой элемент --- изоморфизм, а значит обратим.
\end{enumerate}
\end{proof}
Вернёмся к доказательству единственности. Пусть $V \simeq M_1^{r_1} \oplus M_2^{r_2} \oplus \ldots \oplus M_k^{r_k}$ и $V \simeq N_1^{s_1} \oplus N_2^{s_2} \oplus \ldots \oplus N_l^{s_l}$, где $M_i \not \simeq M_j$ и $N_i \not \simeq N_j$ при разлчиных $i, j$.

Пусть $M_i \not \simeq N_j$. Тогда гомоморфизм $M_i^{r_i} \xrightarrow{\varphi} N_j^{r_j}$ определённый понятным образом, нулевой, по лемме Шура. 

Пусть теперь $M_i \simeq N_j$. Из вышесказанного $\varphi$, очевидно, инъективен, но таким же образом определено и отображение в обратную сторону, но тогда размернсти $M_i^{r_i}$ и $N_j^{s_j}$ совпадают, в частности, из соображений размерности, $r_i = s_j$.

\begin{definition}
Кольцо $R$ называется \textit{$k$-алгеброй}, если $R$ - векторное пространоство над $k$, причём $\lambda(xy) = (\lambda x)y = x(\lambda y)$.
\end{definition}

Из теоремы выше, $k[G] \simeq L_1^{r_1} \oplus L_2^{r_2} \oplus \ldots \oplus L_n^{r_n}$, $L_i$ --- простые и $L_i \not \simeq L_j$, при $i \neq j$.

\begin{proposition}
Пусть $L$ --- простой идеал в $R$. $M$ --- простой модуль. Тогда если $L \not \simeq M$, то $LM = 0$.
\end{proposition}
\begin{proof}
Для любого $m$ определено отображение $l \mapsto lm$. По лемме Шура, все такие отображения нулевые, т. е. $LM = 0$.
\end{proof}

Пусть $1 = e_1 + e_2 + \ldots + e_m$, где $e_i \in L_i^{r_i}$. Отсюда сразу же 
$e_ie_j = 0$ при $i \neq j$, и $e_i^2 = e_i$. Кроме того, очевидно, $e_i \in Z(k[G])$.
Если $a_1e_1 + a_2e_2 + \ldots + a_ne_n = 0$, домножая на $e_i$ получаем $a_i = 0$. Значит $m \leq \dim_kZ(k[G])$.
\begin{theorem}
Пусть $M$ --- простой $G$-модуль. Тогда для какого-то $i$, $L_i \simeq M$.
\end{theorem}
\begin{proof}
Для какого-то $i$, $L_iM \neq 0$. Отсюда $L_i \simeq M$.
\end{proof}

Для $g \in G$ рассмотрим $K_g = \{hgh^{-1} \mid h \in G\}$.
Обозначим $\hat{K}_g = \sum\limits_{\sigma \in K_g}\sigma$.
\begin{theorem}
$\{\hat{K}_g\}_{g \in I}$, где $I$ --- множество представителей классов сопряжённости, составляют базис $Z(k[G])$.
\end{theorem}
$\hat{K}_g$, очевидно, лежит в центре при любом $g \in G$. С другой стороны, если
$x = \sum\limits_{\sigma \in G} a_\sigma \sigma \in Z(k[G])$,
$x = \sum\limits_{\sigma \in G} a_\sigma \tau \sigma \tau^{-1}$, для любого $\tau \in G$, т. е. $a_{\tau \sigma \tau^{-1}} = a_\sigma$. Завершить доказательство можно индукцией по числу ненулевых $a_\sigma$.

\begin{remark}
    Пусть $V$ - одномерное представление. Тогда $V = \langle v \rangle$, где $v \neq 0$, значит, все вектора в $V$ коллинеарны $v$ (т.е. отличаются от $v$ домножением на элемент из $k^*$). Тогда $\forall g \in G \,\exists \chi(g) \in k^* (gv = \chi(g)v)$. Функция $\chi: G \to k^*$ корректно определена, так как значение $\chi(g)$ не зависит от выбора $v$:
    \[
        \chi_{\alpha v}(g)(\alpha v) = g(\alpha v) = \alpha (g v) = \alpha (\chi_v(g) v) = \chi_v(g)(\alpha v) \Ra \chi_{\alpha v} = \chi_v
    \]
    Также верно, что:
    \[
        \chi(g_1 g_2) v = (g_1 g_2) v = g_1 (g_2 v) = g_1 (\chi(g_2) v) = \chi(g_2) (g_1 v) = \chi(g_2) (\chi(g_1) v) \Ra
    \]
    \[
        \Ra \chi(g_1 g_2) = \chi(g_1) \chi(g_2)
    \]
    то есть $\chi$ --- гомоморфизм. Таким образом, одномерным представлениям можно сопоставить гомоморфизм $\chi: G \to k^*$.
\end{remark}

\begin{proposition}
    Пусть представлениям $V, V'$ сопоставляются гомоморфизмы $\chi, \chi'$ соответственно, и $V \overset{f}{\simeq} V'$. Тогда $\chi = \chi'$.
\end{proposition}
\begin{proof}
    Из определения изоморфизма:
    \begin{gather*}
        \forall v \in V, g \in G (f(g v) = g f(v)) \\
        \forall v \in V, \alpha \in k^* (f(\alpha v) = \alpha f(v))
    \end{gather*}
    тогда $\chi(g) f(v) = f(\chi(g) v) = f(g v) = g f(v) = \chi'(g) f(v) \Ra \chi = \chi'$.
\end{proof}

\begin{example}
    $|G| = n$ --- нечётное, $k = \R$, $\chi: G \to \R^*$. Тогда:
    \[
        g \in G \Ra g^n = 1 \Ra \chi(g)^n = 1 \Ra \chi(g) = 1
    \]
    В случае $k = \C$ можно заключить, что $\chi(g) \in \sqrt[n]{1}$.
\end{example}

\begin{example}
    $k = \overline{k}$ (т.е. $k$ --- алгебраически замкнутое поле), $|G| = n$, $G = \langle \sigma \rangle$ --- циклическая группа. Поскольку $\sigma^n = 1$, при любом гомоморфизме $G \to k^*$ $\sigma \mapsto \sqrt[n]{1}$, значит, сигнатурой гомоморфизма можно считать $G = \langle \sigma \rangle \to \mu_n$ (где $\mu_n = \langle a \rangle$ - циклическая группа порядка $n$). Так как образ $\sigma$ задаёт весь гомоморфизм, все возможные гомоморфизмы соответствуют степеням $a$: $\sigma \overset{\varphi_i}{\longmapsto} a^i$. Эти гомоморфизмы образуют группу: $\sigma \overset{\varphi_i \varphi_j}{\longmapsto} a^{i + j}$, причём эта группа циклическая и порождена $\varphi_1$.
\end{example}

\begin{remark}
    Из вышесказанного следует, что если $G$ --- циклическая, $k = \overline{k}$, то $G \simeq X_1(G)$, где $X_1(G)$ - группа характеров $G$, т.е. гомоморфизмов $G \to k^*$.
\end{remark}

\begin{remark}
    Пусть $k = \overline{k}$. Если $G$ --- конечная абелева группа, то, как известно, $G \simeq C_1 \times \hdots \times C_m$, где $\{C_i\}$ --- циклические группы. Тогда:
    \[
        X_1(G) \simeq X_1(C_1 \times \hdots \times C_m) \simeq \prod_{i = 1}^m X_1(C_i) \simeq \prod_{i = 1}^m C_i \simeq G
    \]
\end{remark}

\begin{example}
    $G = S_3$. В $S_3$ есть три класса сопряжённых элементов:
    \begin{enumerate}
        \item $\{\id\}$
        \item $\{(12), (23), (13)\}$
        \item $\{(123), (132)\}$
    \end{enumerate}
    Значит, существует не более чем 3 неприводимых представления.

    Два одномерных представления можно задать гомоморфизмами $G \to k^*$: $\chi_1 = \id$, $\chi_2 = \sgn$ --- знак перестановки.

    Последнее представление существует и является двумерным.
    \begin{equation*}
        U = \langle e_1, e_2, e_3 \rangle,\, V = \{ \lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3 \,|\, \lambda_1 + \lambda_2 + \lambda_3 = 0 \} \subset U,\, \dim(V) = 2
    \end{equation*}
    Чтобы $V$ являлось представлением, нужно задать на $V$ структуру $G$-модуля, то есть определить действие элементов $G$ на $V$. Сделать это несложно:
    \begin{equation*}
        \sigma \in S_3 \Ra \sigma(\lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3) = \lambda_1 e_{\sigma(1)} + \lambda_2 e_{\sigma(2)} + \lambda_3 e_{\sigma(3)}
    \end{equation*}
    Поскольку при таком действии $\lambda_i$ не меняются, не меняется и их сумма, а значит, $\sigma(V) \subset V$.
    
    Осталось показать, что $V$ --- неприводимое. Предположим противное, тогда $V = W \oplus V'$, где $W \neq 0, V \Ra 0 < \dim(W) < \dim(V) \Ra \dim(W) = 1$. Тогда $W = \langle \lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3 \rangle$, где $\lambda_1 + \lambda_2 + \lambda_3 = 0$. Поскольку $\dim(W) = 1$, любые два вектора в $W$ коллинеарны, в частности, для любого $\sigma \in S_3$ $\sigma(\sum \lambda_i e_i) = \alpha \sum \lambda_i e_i$. Но при этом
    \begin{equation*}
        \sigma(\lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3) = \lambda_1 e_{\sigma(1)} + \lambda_2 e_{\sigma(2)} + \lambda_3 e_{\sigma(3)} = \lambda_{\sigma^{-1}(1)} e_1 + \lambda_{\sigma^{-1}(2)} e_2 + \lambda_{\sigma^{-1}(3)} e_3
    \end{equation*}
    
    Обозначим $\sigma^{-1}(1) = \tau$. Поскольку равенство $\sigma \in S_3$ $\sigma(\sum \lambda_i e_i) = \alpha \sum \lambda_i e_i$ должно выполняться для любого $\sigma \in S_3$, система
    \begin{equation*}
        \begin{cases*}
            \alpha \lambda_1 = \lambda_{\tau(1)} \\
            \alpha \lambda_2 = \lambda_{\tau(2)} \\
            \alpha \lambda_3 = \lambda_{\tau(3)}
        \end{cases*}
    \end{equation*}
    при фиксированных $\lambda_i$ должна выполняться для любого $\tau$. Несложно видеть, что ненулевых таких $\lambda_i$ не существует. Поскольку $W = \langle \lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3 \rangle \neq 0$, получили противоречие, значит, $V$ --- неприводимое представление. % пояснить, почему таких \lambda_i не существует
\end{example}

    Аналогичное рассуждение можно проделать и для $N \neq 3$. Для доказательства неприводимости $V$ нужно доказать, что в любом элементе $m \in V, m = \sum \lambda_i e_i, \sum \lambda_i = 0$ все перестановки индексов порождают $V$. \begin{definition}
    Пусть $V$ --- $G$-модуль, $\varphi: G \to \Aut(V)$, $\varphi(g) = (v \mapsto gv) = \varphi_g$. Тогда \textit{характером представления $G$ на $V$} называется $\chi_V(g) = \Tr(\varphi_g)$.
\end{definition}

\begin{proposition}
    $M, N$ --- $G$-модули, $M \simeq N$. Тогда $\chi_M(g) = \chi_N(g)$.
\end{proposition}
\begin{proof}
    % тут диаграммку забабахать
    \begin{equation*}
        \Tr(M \overset{\cdot g}{\to} M) = \Tr(M \overset{\sim}{\to} N \overset{\cdot g}{\to} N \overset{\sim}{\to} M) = A \Tr(N \overset{\cdot g}{\to} N) A^{-1} = \Tr(N \overset{\cdot g}{\to} N)
    \end{equation*}
\end{proof}
