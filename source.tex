\begin{theorem}[Машке]
$k$ --- поле, $G$ --- конечная группа, $\chr k \nmid |G|$, $V$ --- $k[G]$-модуль, $U$ -- подмодуль. Тогда найдётся подмодуль $W$, т. ч.  $V = U \oplus W$.
\end{theorem}

\begin{definition}
\textit{Неприводимый $G$-модуль} --- простой $k[G]$-модуль.
\end{definition}

\begin{definition}
Модуль $M$ называется \textit{неразложимым}, если равенство $M = M_1 \oplus M_2$, означает, что $M_1$ или $M_2$ тривиален.
\end{definition}

Везде далее $V$ --- конечномерный(как векторное пространство над k) $k[G]$-модуль, а также $\chr k \nmid |G|$. В таких условиях, из теоремы Машке, получаем:

\begin{corollary}
     Если $V$ приводим, найдутся ненулевые подмодули $V', V''$, т. ч. $V = V' \oplus V''$.
\end{corollary}

Отсда, индукцией по размерности, сразу получаем существование разложения $V$ в сумму неприводимых.

\begin{lemma}[Шур]
Пусть $M, N$ --- простые $R$-модули. Тогда
\begin{enumerate}
\item Любой ненулевой гомоморфизм $M \xrightarrow{\varphi} N$ является изоморфизмом.
\item $\End_k(M)$ --- тело (т. е. кольцо с делением).
\end{enumerate}
\end{lemma}
\begin{proof} \quad
\begin{enumerate}
\item Пусть $\varphi$ не изоморфизм. Тогда $\ker \varphi \neq 0$, или $\im \varphi \neq N$. В первом случае $\ker \varphi = M$, во втором --- $\im \varphi = 0$, т. е. $\varphi = 0$.
\item Ну действительно, любой ненулевой элемент --- изоморфизм, а значит обратим.
\end{enumerate}
\end{proof}
Рассмотрим теперь вопрос единственности. Пусть $V \simeq M_1^{r_1} \oplus M_2^{r_2} \oplus \ldots \oplus M_k^{r_k}$ и $V \simeq N_1^{s_1} \oplus N_2^{s_2} \oplus \ldots \oplus N_l^{s_l}$, где $M_i \not \simeq M_j$ и $N_i \not \simeq N_j$ при разлчиных $i, j$.

Пусть $M_i \not \simeq N_j$. Тогда гомоморфизм $M_i^{r_i} \xrightarrow{\varphi} N_j^{r_j}$ определённый понятным образом, нулевой, по лемме Шура. 

Пусть теперь $M_i \simeq N_j$. Из вышесказанного $\varphi$, очевидно, инъективен, но таким же образом определено и отображение в обратную сторону, но тогда размерности $M_i^{r_i}$ и $N_j^{s_j}$ совпадают, в частности, из соображений размерности, $r_i = s_j$.

\begin{definition}
Кольцо $R$ называется \textit{$k$-алгеброй}, если $R$ - векторное пространство над $k$, причём $\lambda(xy) = (\lambda x)y = x(\lambda y)$.
\end{definition}

Из теоремы выше, $k[G] \simeq L_1^{r_1} \oplus L_2^{r_2} \oplus \ldots \oplus L_m^{r_m}$, $L_i$ --- простые и $L_i \not \simeq L_j$, при $i \neq j$.

\begin{proposition}
Пусть $L$ --- простой идеал в $R$. $M$ --- простой модуль. Тогда если $L \not \simeq M$, то $LM = 0$.
\end{proposition}
\begin{proof}
Для любого $m$ определено отображение $l \mapsto lm$. По лемме Шура, все такие отображения нулевые, т. е. $LM = 0$.
\end{proof}

\begin{remark}
    Пусть $V$ - одномерное представление. Тогда $V = \langle v \rangle$, где $v \neq 0$, значит, все вектора в $V$ коллинеарны $v$ (т.е. отличаются от $v$ домножением на элемент из $k^*$). Тогда $\forall g \in G \,\exists \chi(g) \in k^* (gv = \chi(g)v)$. Функция $\chi: G \to k^*$ корректно определена, так как значение $\chi(g)$ не зависит от выбора $v$:
    \[
        \chi_{\alpha v}(g)(\alpha v) = g(\alpha v) = \alpha (g v) = \alpha (\chi_v(g) v) = \chi_v(g)(\alpha v) \Ra \chi_{\alpha v} = \chi_v
    \]
    Также верно, что:
    \[
        \chi(g_1 g_2) v = (g_1 g_2) v = g_1 (g_2 v) = g_1 (\chi(g_2) v) = \chi(g_2) (g_1 v) = \chi(g_2) (\chi(g_1) v) \Ra
    \]
    \[
        \Ra \chi(g_1 g_2) = \chi(g_1) \chi(g_2)
    \]
    то есть $\chi$ --- гомоморфизм. Таким образом, одномерным представлениям можно сопоставить гомоморфизм $\chi: G \to k^*$.
\end{remark}

\begin{proposition}
    Пусть представлениям $V, V'$ сопоставляются гомоморфизмы $\chi, \chi'$ соответственно, и $V \overset{f}{\simeq} V'$. Тогда $\chi = \chi'$.
\end{proposition}
\begin{proof}
    Из определения изоморфизма:
    \begin{gather*}
        \forall v \in V, g \in G (f(g v) = g f(v)) \\
        \forall v \in V, \alpha \in k^* (f(\alpha v) = \alpha f(v))
    \end{gather*}
    тогда $\chi(g) f(v) = f(\chi(g) v) = f(g v) = g f(v) = \chi'(g) f(v) \Ra \chi = \chi'$.
\end{proof}

    $|G| = n$ --- нечётное, $k = \R$, $\chi: G \to \R^*$. Тогда:
    \[
        g \in G \Ra g^n = 1 \Ra \chi(g)^n = 1 \Ra \chi(g) = 1

    В случае $k = \C$ можно заключить, что $\chi(g) \in \sqrt[n]{1}$.

\begin{example}
    $k = \overline{k}$ (т.е. $k$ --- алгебраически замкнутое поле), $|G| = n$, $G = \langle \sigma \rangle$ --- циклическая группа. Поскольку $\sigma^n = 1$, при любом гомоморфизме $G \to k^*$ $\sigma \mapsto \sqrt[n]{1}$, значит, сигнатурой гомоморфизма можно считать $G = \langle \sigma \rangle \to \mu_n$ (где $\mu_n = \langle a \rangle$ - циклическая группа порядка $n$). Так как образ $\sigma$ задаёт весь гомоморфизм, все возможные гомоморфизмы соответствуют степеням $a$: $\sigma \overset{\varphi_i}{\longmapsto} a^i$. Эти гомоморфизмы образуют группу: $\sigma \overset{\varphi_i \varphi_j}{\longmapsto} a^{i + j}$, причём эта группа циклическая и порождена $\varphi_1$.
\end{example}

\begin{remark}
    Из вышесказанного следует, что если $G$ --- циклическая, $k = \overline{k}$, то $G \simeq X_1(G)$, где $X_1(G)$ - группа характеров $G$, т.е. гомоморфизмов $G \to k^*$.
\end{remark}

\begin{remark}
    Пусть $k = \overline{k}$. Если $G$ --- конечная абелева группа, то, как известно, $G \simeq C_1 \times \hdots \times C_m$, где $\{C_i\}$ --- циклические группы. Тогда:
    \[
        X_1(G) \simeq X_1(C_1 \times \hdots \times C_m) \simeq \prod_{i = 1}^m X_1(C_i) \simeq \prod_{i = 1}^m C_i \simeq G
    \]
\end{remark}

    \begin{equation*}

    \end{equation*}
    Чтобы $V$ являлось представлением, нужно задать на $V$ структуру $G$-модуля, то есть определить действие элементов $G$ на $V$. Сделать это несложно:
    \begin{equation*}
        \sigma \in S_3 \Ra \sigma(\lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3) = \lambda_1 e_{\sigma(1)} + \lambda_2 e_{\sigma(2)} + \lambda_3 e_{\sigma(3)}
    \end{equation*}
    Поскольку при таком действии $\lambda_i$ не меняются, не меняется и их сумма, а значит, $\sigma(V) \subset V$.
    
    Осталось показать, что $V$ --- неприводимое. Предположим противное, тогда $V = W \oplus V'$, где $W \neq 0, V \Ra 0 < \dim(W) < \dim(V) \Ra \dim(W) = 1$. Тогда $W = \langle \lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3 \rangle$, где $\lambda_1 + \lambda_2 + \lambda_3 = 0$. Поскольку $\dim(W) = 1$, любые два вектора в $W$ коллинеарны, в частности, для любого $\sigma \in S_3$ $\sigma(\sum \lambda_i e_i) = \alpha \sum \lambda_i e_i$. Но при этом
    \begin{equation*}
        \sigma(\lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3) = \lambda_1 e_{\sigma(1)} + \lambda_2 e_{\sigma(2)} + \lambda_3 e_{\sigma(3)} = \lambda_{\sigma^{-1}(1)} e_1 + \lambda_{\sigma^{-1}(2)} e_2 + \lambda_{\sigma^{-1}(3)} e_3
    \end{equation*}
    
    Обозначим $\sigma^{-1}(1) = \tau$. Поскольку равенство $\sigma \in S_3$ $\sigma(\sum \lambda_i e_i) = \alpha \sum \lambda_i e_i$ должно выполняться для любого $\sigma \in S_3$, система
    \begin{equation*}

            \alpha \lambda_1 = \lambda_{\tau(1)} \\
            \alpha \lambda_2 = \lambda_{\tau(2)} \\
            \alpha \lambda_3 = \lambda_{\tau(3)}
        \end{cases*}
    \end{equation*}

Поскольку $W = \langle \lambda_1 e_1 + \lambda_2 e_2 + \lambda_3 e_3 \rangle \neq 0$, получили противоречие, значит, $V$ --- неприводимое представление. % пояснить, почему таких \lambda_i не существует

\begin{proposition}
    $\chi_{M \oplus N} = \chi_M + \chi_N$.
\end{proposition}
\begin{proof}

    \begin{equation*}
        (M \oplus N \overset{\cdot g}{\longrightarrow} M \oplus N) \mapsto \begin{pmatrix}
            (\tilde{M}) & 0 \\
            0 & (\tilde{N})
        \end{pmatrix}
    \end{equation*}
    где $\tilde{M}, \tilde{N}$ - записи операторов умножения на $g$ в $M$ и $N$ в соответствующих базисах. Поскольку получившаяся матрица блочно-диагональна, $\chi_{M \oplus N}(g) = \Tr_{M \oplus N}(\cdot g) = \Tr_M(\cdot g) + \Tr_N(\cdot g) = \chi_M(g) + \chi_N(g)$.
\end{proof}

\begin{theorem}[Второе соотношение ортогональности]
Если $g_1, g_2, \ldots, g_s$ --- представители классов сопряжённости, $m_i$ --- число элементов в классе, то $\sum_k \chi_k(g_i^{-1}) \chi_k(g_j) = \frac{n}{m_j}\delta_{ij}$.
\end{theorem}
\begin{proof}
	Пусть $A = (m_k \chi_i(g_k))$, $B = (\chi_j(g_k^{-1})$. Тогда $(AB)_{ij} = \sum m_k \chi_i(g_k) \chi_j(g_k^{-1})$ = $\sum_C \sum_{g \in C} \chi(g) \chi_j(g^{-1}) = \sum_{g \in G} \chi(g) \chi_j(g^{-1}) = n\delta_{ij}$. Иначе говоря, $AB = nI$, но тогда $BA = nI$, т. е. $n\delta_{ij} = \sum_k \chi_k(g_i^{-1}) m_j \chi_k(g_j)$, или же $\sum_k \chi_k(g_i^{-1}) \chi_k(g_j) = \frac{n}{m_j}\delta_{ij}$. Важно отметить, что $\chr k \nmid m_j$, ведь $m_j$ --- размер орбиты, а значит $m_j = \frac{n}{Z(g_j)}$, где $Z(g_j)$ --- подгруппа элементов, коммутирующих с $g_j$.
\end{proof}

\end{proposition}

\begin{remark}
    Из этого утверждения следует, что $\tilde{\chi}$ не может раскладываться дальше в нетривиальную сумму неприводымых, а значит, сам является неприводимым, а значит, $U$ --- действительно неприводимое представление.

\begin{proof}
    Пусть $g \in G = S_n$. Обозначим за $N(g)$ количество $i \in \{1, \hdots, n\}$ таких, что $g(i) = i$. Поскольку $G$ --- это все перестановки, у всякого элемента $\{1, \hdots, n\}$ орбитой под действием $G$ является всё это множество, то есть всего орбит ровно одна. Тогда, по лемме Бернсайда,
    \begin{equation*}
        \sum_{g \in S_n} N(g) = 1 \cdot |S_n| = n!
    \end{equation*}

    Обозначим $G_i = \{ g\in G \,|\, g(i) = i \} \subset G$. Для удобства рассмотрим $G_n$ --- эта подгруппа действует перстановками на $\{1, \hdots, n-1\}$. Аналогично, для $g \in G_n$ обозначим за $N'(g)$ количество $i \in \{1, \hdots, n-1\}$ таких, что $g(i) = i$. Из определений очевидно, что $N'(g) = N(g) - 1$.

    Опять же, по лемме Бернсайда,
    \begin{equation*}
        \sum_{g \in G_n} N'(g) = \sum_{g \in G_n} (N(g) - 1) = |G_n| \Ra \sum_{g \in G_n} N(g) = 2 \cdot |G_n| = 2 \cdot (n-1)!
    \end{equation*}
    Аналогичное равенство верно для всякой подгруппы $G_i$.
    \begin{equation*}
        \sum_{i = 1}^{n} \sum_{g \in G_i} N(g) = 2 \sum_{i = 1}^{n} |G_i| = 2 \cdot n \cdot |G_i| = 2 \cdot n \cdot (n-1)! = 2 \cdot n! = 2 \cdot |G|
    \end{equation*}
    Внимательно рассмотрим двойную сумму в левой части равенства. $g \in G_i$ по определению означает, что $g(i) = i$. Тогда, как следует из определения $N(g)$, для всякого элемента $g \in G$ условие $g \in G_i$ выполнено в точности для $N(g)$ разных $G_i$. Значит, в двойной сумме каждый элемент $g$ был учтён $N(g)$ раз. Таким образом, эту двойную сумму можно переписать как
    \begin{equation*}
        \sum_{i = 1}^{n} \sum_{g \in G_i} N(g) = \sum_{g \in \bigcup G_i} N(g) \cdot N(g) = \sum_{g \in \bigcup G_i} N(g)^2 = 2 |G|
    \end{equation*}
    Заметим, что несмотря на то, что $\bigcup G_i \neq G$, если $g \notin \bigcup G_i$, то $N(g) = 0$, а значит, суммировать можно и по всей группе $G$:
    \begin{equation*}
        \sum_{g \in G} N(g)^2 = 2 |G| \Ra \frac{1}{|G|} \sum_{g \in G} N(g)^2 = 2
    \end{equation*}

    Вернёмся к $\chi_\Phi$. Умножение на $g \in G$ действует на $V$ перестановкой базисных векторов, значит, матрица этого умножения --- это единичная матрица с переставленными соответствующим образом столбцами. Но тогда след этой матрицы (то есть $\chi_\Phi(g)$) равен количеству единиц на диагонали, а они соответствуют неподвижным при умножении $g$ базисным элементам, то есть $i$ таким, что $g(i) = i$. Итого, $\chi_\Phi(g) = N(g)$. Дополнительно отметим, что $N(g) = N(g^{-1})$.
    \begin{equation*}
        \langle \chi_\Phi, \chi_\Phi \rangle = \frac{1}{|G|} \sum_{g \in G} \chi_\Phi(g) \chi_\Phi(g^{-1}) = \frac{1}{|G|} \sum_{g \in G} N(g)^2 = 2
    \end{equation*}
\end{proof}

\begin{remark}

        &= \frac{1}{|G| \cdot |H|} \sum_{(g, h) \in G \times H} \chi_1(g) \rho_1(h) \chi_2(g^{-1}) \rho_2(h^{-1}) \\
        &= \frac{1}{|G| \cdot |H|} \left( \sum_{g \in G} \chi_1(g) \chi_2(g^{-1}) \right) \left( \sum_{h \in H} \rho_1(h) \rho_2(h^{-1}) \right) \\
        &= \langle \chi_1, \chi_2 \rangle \langle \rho_1, \rho_2 \rangle
    \end{align*}

    В частности, $\langle \chi \boxtimes \rho \rangle = \langle \chi, \chi \rangle \langle \rho, \rho \rangle$. Таким образом, если $\chi, \rho$ --- неприводимые характеры, то их скалярные квадраты равны 1, тогда
    \begin{equation*}
        \langle \chi \boxtimes \rho \rangle = \langle \chi, \chi \rangle \langle \rho, \rho \rangle = 1 \cdot 1 = 1
    \end{equation*}
    а значит, $\chi \boxtimes \rho$ --- тоже неприводимый характер.
\end{remark}

\begin{theorem}

\end{theorem}
\begin{proof}
	$\C[G] \simeq \C^{\oplus n}$, пусть $\{e_i\}_{i = 1}^n$ --- центральные идемпотентны. Пусть $a = \sum\limits_{\sigma \in G} \alpha_\sigma \sigma = \sum\limits_{i = 1}^n \beta_i e_i$, но у $a$ конечный порядок, обозначим его $m$, а значит $\beta_i^m = 1$. Пусть $\tau$ --- произвольный элемент группы, т. ч. $a_\tau \neq 0$. $\sum\limits_{\sigma \in G} a_\sigma \sigma \tau^{-1}  = \sum\limits_{i = 1}^n \beta_i e_i \tau^{-1}$. Применив регулярный характер, получаем $na_\tau = \sum\limits_{i = 1}^n \beta_i \tau^{-1} \chi_{reg}(e_i\tau^{-1}) = \sum\limits_{i = 1}^n \beta_i \chi_i(\tau^{-1})$. Но $|\beta_i \chi_i(\tau^{-1})| =  1$, значит $\beta_i \chi_i(\tau^{-1}) = \beta_j \chi_j(\tau^{-1}) = a_\tau = \pm 1$, т. е. $\beta_i = \pm \chi_i(\tau)$(знак не зависит от $i$). Т. е., т. к. $\{e_i\}$ --- базис, $a = \pm \tau$.
\end{proof}

\begin{remark}

\begin{lemma}[Нётер о нормализации]
    Пусть $A$ --- конечнопорождённая $k$-алгебра. Тогда найдутся алгебраически независимые $y_1, \hdots, y_l \in A$ (возможно, $l = 0$) такие, что $A$ --- конечнопорождённый модуль над $k[y_1, \hdots, y_l]$.
\end{lemma}
\begin{proof}
    $A = k[x_1, \hdots, x_n]$. Проведём индукцию по $n$. Если $n = 1$, то $A = k[x_1]$. В случае, если $x_1$ алгебраичен над $k$, то $A$ --- конечнопорождённый модуль над $k$ (порождается конечным числом степеней $x_1$). Если же $x_1$ не алгебраичен над $k$, то можно взять $y_1 = x_1$, и тогда $A = k[y_1]$.

    Индукционнный переход: пусть $A = k[x_1, \hdots, x_n]$. Если набор $\{x_1, \hdots, x_n\}$ алгебраически независим, то можно взять $y_i = x_i$. Иначе перенумеруем $x_1, \hdots, x_n$ таким образом, что $x_1, \hdots, x_r$ алгебраически независимы, а $x_{r+1}, \hdots, x_n$ алгебраически зависят от них; $r < n$.

    Распишем определение алгебраической зависимости $x_1, \hdots, x_n$: найдтся полином $f \not\equiv 0$ такой, что $f(x_1, \hdots, x_n) = 0$. Выделим из него степени $x_n$ и рассмотрим старшую из них:
    \begin{equation*}
        f(x_1, \hdots, x_n) = a(x_1, \hdots, x_{n-1}) \cdot x_n^s + \hdots = 0
    \end{equation*}
    Разберём два случая.

    Если $a(x_1, \hdots, x_{n-1}) = const \in k$, то воспользуемся индукционным предположением для $n-1$:
    \begin{equation*}
        k \subset k[y_1, \hdots, y_l] \subset k[x_1, \hdots, x_{n-1}] \subset k[x_1, \hdots, x_n]
    \end{equation*}
    где $k[x_1, \hdots, x_n]$ является конечнопорождённым $k[x_1, \hdots, x_{n-1}]$-модулем, поскольку порождается степенями $x_n$ от 1 до $s-1$.
    
    При этом $k[x_1, \hdots, x_{n-1}]$ есть конечнопорождённый $k[y_1, \hdots, y_l]$-модуль по индукционному предположению. Значит, $k[x_1, \hdots, x_n]$ --- конечнопорождённый $k[y_1, \hdots, y_l]$-модуль, и $y_1, \hdots, y_l$ алгебраически независимы. Таким образом, в данном случае лемма доказана.

    Если же $a(x_1, \hdots, x_{n-1})$ не является константой, то можно сделать следующую замену переменных:
    \begin{equation*}
        x_n' = x_n, \quad \forall i \in \{ 1, \hdots, n-1 \} \left( x_i' = x_i - x_n^{(d^i)} \right)
    \end{equation*}
    где $d \in \N$ мы выберем позднее. Понятно, что $$k[x_1, \hdots, x_{n-1}, x_n] = k[x_1', \hdots, x_{n-1}', x_n]$$

    Как и раньше, рассмотрим степени $x_n$ в $f(x_1, \hdots, x_n)$:
    \begin{align*}
        0 &= f(x_1, \hdots, x_n) \\
        &= \sum a_{i_1, \hdots, i_n} x_1^{i_1} \hdots x_n^{i_n} \\
        &= \sum a_{i_1, \hdots, i_n} \prod_{k = 1}^{n - 1} \left( x_k' + x_n^{d^k} \right) \cdot x_n^{i_n}
    \end{align*}
    В каждом из слагаемых при раскрытии скобок максимальная степень $x_n$ равна $i_n + i_1 d + i_2 d^2 + \hdots + i_{n-1} d^{n-1}$. Если мы выберем такое $d$, что все такие степени различны, то перед максимальной степень во всей сумме будет стоять коэффициент, равный соответствующему $a_{\hdots}$. Таким образом, доказательство сведётся к уже разобранному случаю.

    Осталось только выбрать подходящее $d$. Подходит любое $d > \max i_k$ (максимум берётся среди всех возможможных наборов индексов), т.к. в таком случае сумма $i_n + i_1 d + i_2 d^2 + \hdots + i_{n-1} d^{n-1}$ есть в точности $d$-ичная запись некоторого числа. Известно, что у любого числа есть ровно одна такая запись, значит, значения таких сумм не могут совпадать.
\end{proof}

\begin{proposition}
    Пусть $k$ --- поле, $A$ --- конечнопорождённая $k$-алгебра, и $A$ --- поле. Тогда $A$ --- конечное расширение $k$.
\end{proposition}
\begin{proof}
    $k \subset k[t_1, \hdots, t_n] \subset A$. Предположим, что утверждение не выполнено, то есть $n > 0$.

    Расмотрим идеал $I \subset k[t_1, \hdots, t_n]$, $I \neq (0), (1)$. Поскольку $A$ --- поле,
    \begin{equation*}
        IA = \left\{ \sum_{j = 1}^{m} i_j a_j \big| i_j \in I, a_j \in A, m \in \N \right\} = A
    \end{equation*}
    Рассмотрим элементы $\{ u_1, \hdots, u_n \}$, которые порождают $A$ как $k[t_1, \hdots, t_n]$-модуль. $IA = A$, поэтому каждое $u_i$ представляется в виде конечной суммы $u_i = \sum a_{ij} v_{ij}$, $a_{ij} \in I, v_{ij} \in A$.

    В то же время $\{u_1, \hdots, u_n\}$ порождают всё $A$, в том числе и элементы $v_{ij}$.
    \begin{equation*}
        u_i = \sum a_{ij} v_{ij} = \sum a_{ij} \left( \sum b_{kj} u_{k} \right) \Ra u_i = \sum d_{ij} u_j, \quad d_{ij} \in I \subset k[t_1, \hdots, t_n]
    \end{equation*}
    Итого, получаем равенство
    \begin{equation*}
        \begin{pmatrix}
            1 - d_{11} & -d_{12} & \cdots & -d_{1n} \\
            -d_{21} & 1 - d_{22} & \cdots & -d_{2n} \\
            \vdots & \vdots & \ddots & \vdots \\
            -d_{n1} & -d_{n2} & \cdots & 1 - d_{nn}
        \end{pmatrix}
        \begin{pmatrix}
            u_1 \\ u_2 \\ \vdots \\ u_n
        \end{pmatrix}
        = M u = 0
    \end{equation*}
    Обозначим $D = \det M$. С одной стороны, $M u = 0 \Ra D = 0$. С другой стороны, если расписать определитель по определению, то получится, что $D = 1 + S$, где $S \in I$. $0 = D = 1 + S \Ra S = -1 \in I$, что приводит к противоречию, ведь $I \neq (1)$. 
\end{proof}

    Начнём делить $g$ с остатком поочерёдно на все $t_i - a_i$. Понятно, что после каждого деления в остатке будет отсутствовать соответствующая переменная $t_i$. Значит, в конце получится константа; она, очевидно, будет равна $g(a_1, \hdots, a_n)$. Если $g(a_1, \hdots, a_n) = 0$, то $g$ уже лежит в нашем идеале. Если же $g(a_1, \hdots, a_n) \neq 0$, то
    \begin{equation*}
        1 = \frac{g(a_1, \hdots, a_n)}{g(a_1, \hdots, a_n)} \in (t_1 - a_1, \hdots, t_n - a_n, g)
    \end{equation*}
    то есть добавить $g$ с сохранением собственности идеала не получится.

    $\Ra$: Если $m$ максимален, то $k \subset k[t_1, \hdots, t_n] / m = A$ --- поле. При этом $A$ конечно порождается $\overline{t_i}$ как $k$-алгебра. Тогда, по условно доказанному утверждению выше, $A / k$ --- конечное расширение, и, поскольку $k = \overline{k}$, $A = k$. Но если $A = k$, то
    \begin{equation*}
        \forall \overline{t_i} \exists a_i \in A = k \left( \overline{t_i} - a_i = 0 \right) \Ra t_i - a_i \in m \Ra (t_1 - a_1, \hdots, t_n - a_n) \subset m
    \end{equation*}
    Мы уже знаем, что идеалы такого вида максимальны, а значит, $(t_1 - a_1, \hdots, t_n - a_n) = m$.
\end{proof}

\begin{proposition}
    $R = k [t_1, \hdots, t_n]$, $f_1, \hdots, f_m \in R$. Тогда следующие утверждения эвивалентны:
    \begin{enumerate}
        \item $(f_1, \hdots, f_n) = (1)$
        \item Система $\forall i \in \{1, \hdots, m\} (f_i(t_1, \hdots, t_n) = 0)$ не имеет решений
    \end{enumerate}
\end{proposition}
\begin{proof}
    $(1) \Ra (2)$: Если бы решение такой системы существовало, то при подстановке этого решения в выражение 1 через $f_1, \hdots, f_m$ получилось бы равенство $0 = 1$.

    $(2) \Ra (1)$: Если $(f_1, \hdots, f_n) \neq (1)$, то $(f_1, \hdots, f_n)$ можно дополнить до максимального идеала $m$. Но как мы знаем, все максимальные идеалы в $R$ имеют вид $(t_1 - a_1,\hdots, t_n - a_n)$. При подстановке $a_1, \hdots, a_n$ в $f_i$ получаем 0, т.к. $f_i \in (t_1 - a_1,\hdots, t_n - a_n) = m$, то есть у системы есть решение.
\end{proof}

\end{equation*}
Заметим, что при раскрытии скобок слагаемые с пересекающимимся наборами индексов обнулятся. Тогда два набора индексов можно объединить в один набор размера $n$. Поскольку всего базисных векторов $n$, второй набор в таком случае будет в точности дополнением первого. Красткости ради обозначим набор $i_1 < \hdots , i_k$ как $I$, а множество $\{1, \hdots, n\}$ как $[n]$.
\begin{equation*}
    u_1 \wedge \hdots \wedge u_k \wedge v_1 \wedge \hdots \wedge v_{n-k} = \sum_{\substack{I \subset [n] \\ |I| = k}} \det(B^I) \det(C^{[n] \setminus I}) \cdot e_I \wedge e_{[n] \setminus I} \cdot (-1)^{\sigma(I, [n] \setminus I)}
\end{equation*}
В этом равенстве $[n] \setminus I$ рассматривается как упорядоченный набор, а $\sigma(I, [n] \setminus I)$ означает знак перестановки. Множитель $(-1)^{\sigma(I, [n] \setminus I)}$ появляется, потому что, хотя наборы $I$, $[n] \setminus I$ упорядочены сами по себе, они не упорядочены в совокупности, из-за чего в $e_I \wedge e_{[n] \setminus I}$ базисные вектора могут стоять в неправильном порядке.

Выясним, чему же равно $\sigma(I, [n] \setminus I)$. Для этого вычислим количество инверсий. Поскольку наборы $I$, $[n] \setminus I$ упорядочены сами по себе, все инверсии в перестановке $I, [n] \setminus I$ имеют вид $I \ni a > b \in [n] \setminus I$. Потенциально, инверсиями с элементом $i_s \in I$ могут быть все пары $(i_s, 1), (i_s, 2), \hdots, (i_s, i_s - 1)$. Но некоторые вторые элементы из этих пар также лежат в $I$, и потому инверсий не образуют; таких элементов, поскольку $I$ упорядочено, ровно $s - 1$. Таким образом, количество инверсий с элементом $i_s$ равно $(i_s - 1) - (s - 1) = i_s - s$.
\begin{gather*}
    \sigma(I, [n] - I) = \sum_{s = 1}^{k} i_s - s = \sum_{s = 1}^{k} i_s - \frac{k(k+1)}{2} \\
    \det(A) = \sum_{\substack{I \subset [n] \\ |I| = k}} (-1)^{\sum_{s = 1}^{k} i_s - \frac{k(k+1)}{2}} \det(B^I) \det(C^{[n] \setminus I})
\end{gather*}

Полученное равенство называют \textit{разложением определителя по группе столбцов}.

            U^n \arrow{rr}{f} \arrow{dr}[below left]{\pi} & & V \\
            & \bigwedge^p U \arrow[dashed]{ur}[below right]{! \exists}
        \end{tikzcd}
    \end{equation*}
\end{proposition}
\begin{proof}
    Из универсального свойства тензорного произведения знаем, что
    \begin{equation*}
        \begin{tikzcd}[column sep=small]
            U^n \arrow{rr}{f} \arrow{dr}[below left]{\pi} & & V \\
            & U^{\otimes n} \arrow[dashed]{ur}[below right]{! \exists}
        \end{tikzcd}
    \end{equation*}
    где $\pi((u_1, \hdots, u_n)) = u_1 \otimes \hdots \otimes u_n$.

    Осталось профакторизовать в этой диаграмме тензорное произведение по идеалу, порождённому элементами $\hdots u_i \otimes u_i \hdots$. Для доказательства того, что это можно сделать, рассмотрим элемент такого вида. Т.к. $f$ кососимметрично, $f((\hdots, u_i, u_i, \hdots)) = 0$.

            (\hdots, u_i, u_i, \hdots) \arrow[mapsto]{rr}{f} \arrow[mapsto]{dr}[below left]{\pi} & & 0 \\
            & \hdots \otimes u_i \otimes u_i \otimes \hdots \arrow[dashed, mapsto]{ur}
        \end{tikzcd}
    \end{equation*}
    Значит, отображение, полученное из универсального свойства тензорного произведения, действительно пропускается через факторизацию.
\end{proof}

            w' \wedge w' = 0 \\
            w' \wedge v \wedge e_n = 0
